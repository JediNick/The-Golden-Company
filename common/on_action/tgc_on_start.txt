on_game_start = {
	on_actions = {
		tgc_start
		delay = { days = 1 }
		tgc_artifacts_setup
	}
}

tgc_start = {
	effect = {
       set_global_variable = {
			name = tgc_enabled
			value = yes
		}
       if = {
           limit = {
               exists = global_var:tgc_enabled
			}
		}
	}
}

tgc_artifacts_setup = {
	effect = {
		tgc_artifacts_setup_effect = yes 
	}
}
